import javax.swing.*; import java.awt.*; import java.awt.event.*; public class ButtonCounterGUI extends JFrame { private int count = 0; private JLabel countLabel; public ButtonCounterGUI() { setTitle("Button Counter"); setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); setSize(300, 200); setLayout(new BorderLayout()); countLabel = new JLabel("Count: 0", JLabel.CENTER); countLabel.setFont(new Font("Arial", Font.PLAIN, 24)); add(countLabel, BorderLayout.CENTER); JPanel buttonPanel = new JPanel(); buttonPanel.setLayout(new FlowLayout()); JButton incrementButton = new JButton("Increment"); incrementButton.addActionListener(new ActionListener() { public void actionPerformed(ActionEvent e) { count++; updateCountLabel(); } }); JButton decrementButton = new JButton("Decrement"); decrementButton.addActionListener(new ActionListener() { public void actionPerformed(ActionEvent e) { count--; updateCountLabel(); } }); JButton resetButton = new JButton("Reset"); resetButton.addActionListener(new ActionListener() { public void actionPerformed(ActionEvent e) { count = 0; updateCountLabel(); } }); JButton doubleButton = new JButton("Double"); doubleButton.addActionListener(new ActionListener() { public void actionPerformed(ActionEvent e) { count *= 2; updateCountLabel(); } }); buttonPanel.add(incrementButton); buttonPanel.add(decrementButton); buttonPanel.add(resetButton); buttonPanel.add(doubleButton); add(buttonPanel, BorderLayout.SOUTH); setVisible(true); } private void updateCountLabel() { countLabel.setText("Count: " + count); } public static void main(String[] args) { SwingUtilities.invokeLater(new Runnable() { public void run() { new ButtonCounterGUI(); } }); } }
